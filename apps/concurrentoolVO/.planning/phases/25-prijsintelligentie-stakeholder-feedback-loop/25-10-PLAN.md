```typescript
const handleSave = async (newConfig: PricingStrategy) => {
  if (!activeDbConfig) {
    // No DB config exists yet for this provider — cannot save
    // This can happen if seed migration hasn't run
    console.warn(`[AdminConfigEditor] No DB config found for ${activeTab} — seed migration may not have run`);
    return;
  }

  await updatePricingConfig(activeDbConfig.id, newConfig as unknown as Record<string, unknown>);

  // Reload pricing data store so engine uses updated config
  await usePricingDataStore.getState().loadFromSupabase();

  // Invalidate React Query cache so UI reflects changes
  queryClient.invalidateQueries({ queryKey: ['pricing-configs'] });
};
```

This ensures:
- `updatePricingConfig(id, configData)` is called with the correct DB config ID and the new PricingStrategy object cast to Record<string, unknown> (matching the function signature)
- After successful save, `loadFromSupabase()` reloads the pricing data store so engine calculations use the updated config
- React Query cache for `['pricing-configs']` is invalidated so any components using `usePricingConfigs()` re-fetch
- If no DB config exists (seed migration not run), a warning is logged instead of crashing
- The ProviderConfigForm's try/catch around `onSave` already handles errors with Dutch error messages

Remove the `console.log` and both `// TODO` comment lines entirely.
