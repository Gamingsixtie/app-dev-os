    Create Zod schemas matching each PricingStrategy discriminated union variant:
    ```typescript
    import { z } from 'zod';

    const tierSchema = z.object({
      name: z.string().min(1, 'Naam is verplicht'),
      minStudents: z.number().int().min(0, 'Minimum moet >= 0 zijn'),
      maxStudents: z.number().int().min(1, 'Maximum moet >= 1 zijn'),
      pricePerStudent: z.number().positive('Prijs moet positief zijn'),
    }).refine(d => d.maxStudents > d.minStudents, {
      message: 'Maximum moet groter zijn dan minimum',
    });

    const packageSchema = z.object({
      name: z.string().min(1, 'Naam is verplicht'),
      modules: z.array(z.string()).min(1, 'Minimaal 1 module vereist'),
      pricePerStudent: z.number().positive('Prijs moet positief zijn'),
    });

    const bundleSchema = z.object({
      name: z.string().min(1, 'Naam is verplicht'),
      modules: z.array(z.string()).min(1, 'Minimaal 1 module vereist'),
      discount: z.number().min(0).max(100, 'Korting moet tussen 0 en 100% zijn'),
    });

    const contractPeriodSchema = z.object({
      years: z.number().int().positive(),
      discount: z.number().min(0).max(100),
    });

    export const flatConfigSchema = z.object({ type: z.literal('flat') });
    export const tieredLicenseConfigSchema = z.object({
      type: z.literal('tiered-license'),
      tiers: z.array(tierSchema).min(1, 'Minimaal 1 tier vereist'),
    });
    export const packageBundleConfigSchema = z.object({
      type: z.literal('package-bundle'),
      packages: z.array(packageSchema).min(1, 'Minimaal 1 pakket vereist'),
    });
    export const platformModuleConfigSchema = z.object({
      type: z.literal('platform+module'),
      platformFee: z.number().min(0, 'Platformkosten moeten >= 0 zijn'),
      bundles: z.array(bundleSchema).optional(),
      contractPeriods: z.array(contractPeriodSchema).optional(),
    });

    export const pricingConfigSchema = z.discriminatedUnion('type', [
      flatConfigSchema, tieredLicenseConfigSchema,
      packageBundleConfigSchema, platformModuleConfigSchema,
    ]);
    ```

    Also export a `validatePricingConfig` function that instantiates the appropriate calculator with the proposed config and runs a test calculation. If the calculator throws, the config is invalid (per research pitfall 5):
    ```typescript
    export function validatePricingConfig(
      provider: string,
      config: PricingStrategy
    ): { valid: boolean; error?: string } {
      try {
        // Parse with Zod first
        pricingConfigSchema.parse(config);
        // Then try a test calculation to verify calculator compatibility
        const calculator = createCalculatorForProvider(provider, config);
        calculator.calculate('rekenen', 100); // test with dummy values
        return { valid: true };
      } catch (e) {
        return { valid: false, error: e instanceof Error ? e.message : 'Ongeldige configuratie' };
      }
    }
    ```
