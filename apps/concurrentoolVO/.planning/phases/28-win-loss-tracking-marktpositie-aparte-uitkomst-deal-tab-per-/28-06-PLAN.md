    /** Fetch all discounts for a deal_outcome. */
    export async function fetchDiscounts(dealOutcomeId: string): Promise<DealDiscount[]> {
      const { data, error } = await supabase
        .from('deal_discounts')
        .select('*')
        .eq('deal_outcome_id', dealOutcomeId)
        .order('created_at', { ascending: true });
      if (error) throw error;
      return data.map(mapDealDiscountRow);
    }

    /** Create a discount + write audit-log entry. */
    export async function createDiscount(input: {
      dealOutcomeId: string;
      moduleId: string;
      provider: ProviderKey;
      discountPercentage?: number;
      discountAmount?: number;
    }): Promise<DealDiscount> {
      // Validate XOR client-side (defense — DB CHECK enforces too)
      if ((input.discountPercentage !== undefined) === (input.discountAmount !== undefined)) {
        throw new Error('XOR violation: exactly one of percentage/amount must be set');
      }
      const { data, error } = await supabase
        .from('deal_discounts')
        .insert({
          deal_outcome_id: input.dealOutcomeId,
          module_id: input.moduleId,
          provider: input.provider,
          discount_percentage: input.discountPercentage ?? null,
          discount_amount: input.discountAmount ?? null,
        })
        .select('*')
        .single();
      if (error) throw error;
      const discount = mapDealDiscountRow(data);
      try {
        await supabase.from('deal_audit_log').insert({
          deal_outcome_id: input.dealOutcomeId,
          action: 'discount_added',
          entity_type: 'deal_discount',
          entity_id: discount.id,
          before_value: null,
          after_value: {
            module_id: input.moduleId,
            provider: input.provider,
            discount_percentage: input.discountPercentage ?? null,
            discount_amount: input.discountAmount ?? null,
          },
        });
      } catch (auditErr) {
        console.warn('[deal-outcomes-operations] audit-log write failed', auditErr);
      }
      return discount;
    }

    /** Update an existing discount + write audit-log entry. */
    export async function updateDiscount(id: string, patch: {
      discountPercentage?: number;
      discountAmount?: number;
    }): Promise<DealDiscount> {
      const { data: before } = await supabase.from('deal_discounts').select('*').eq('id', id).single();
      if (!before) throw new Error('Discount not found');
      const { data, error } = await supabase
        .from('deal_discounts')
        .update({
          discount_percentage: patch.discountPercentage ?? null,
          discount_amount: patch.discountAmount ?? null,
        })
        .eq('id', id)
        .select('*')
        .single();
      if (error) throw error;
      try {
        await supabase.from('deal_audit_log').insert({
          deal_outcome_id: before.deal_outcome_id,
          action: 'discount_updated',
          entity_type: 'deal_discount',
          entity_id: id,
          before_value: before,
          after_value: data,
        });
      } catch (auditErr) {
        console.warn(auditErr);
      }
      return mapDealDiscountRow(data);
    }

    /** Delete a discount + write audit-log entry. */
    export async function deleteDiscount(id: string): Promise<void> {
      const { data: before } = await supabase.from('deal_discounts').select('*').eq('id', id).single();
      if (!before) return; // idempotent
      const { error } = await supabase.from('deal_discounts').delete().eq('id', id);
      if (error) throw error;
      try {
        await supabase.from('deal_audit_log').insert({
          deal_outcome_id: before.deal_outcome_id,
          action: 'discount_deleted',
          entity_type: 'deal_discount',
          entity_id: id,
          before_value: before,
          after_value: null,
        });
      } catch (auditErr) {
        console.warn(auditErr);
      }
    }

    /** Read-only audit log entries for a deal. */
    export async function fetchAuditLog(dealOutcomeId: string): Promise<DealAuditEntry[]> {
      const { data, error } = await supabase
        .from('deal_audit_log')
        .select('*')
        .eq('deal_outcome_id', dealOutcomeId)
        .order('timestamp', { ascending: false })
        .limit(100);
      if (error) throw error;
      return data.map(mapDealAuditRow);
    }

    import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
    import { fetchDiscounts, createDiscount, updateDiscount, deleteDiscount } from '@/db/deal-outcomes-operations';

    export function useDealDiscounts(dealOutcomeId: string | undefined) {
      return useQuery({
        queryKey: ['deal-discounts', dealOutcomeId],
        queryFn: () => dealOutcomeId ? fetchDiscounts(dealOutcomeId) : [],
        enabled: !!dealOutcomeId,
        staleTime: 10 * 1000,
      });
    }

    export function useCreateDiscount() {
      const qc = useQueryClient();
      return useMutation({
        mutationFn: createDiscount,
        onSuccess: (d) => {
          qc.invalidateQueries({ queryKey: ['deal-discounts', d.dealOutcomeId] });
          qc.invalidateQueries({ queryKey: ['deal-audit-log', d.dealOutcomeId] });
        },
      });
    }
    export function useUpdateDiscount() {
      const qc = useQueryClient();
      return useMutation({
        mutationFn: ({ id, patch }: { id: string; patch: { discountPercentage?: number; discountAmount?: number } }) =>
          updateDiscount(id, patch),
        // Pessimistic per D-14 — no optimistic update
        onSuccess: (d) => {
          qc.invalidateQueries({ queryKey: ['deal-discounts', d.dealOutcomeId] });
          qc.invalidateQueries({ queryKey: ['deal-audit-log', d.dealOutcomeId] });
        },
      });
    }
    export function useDeleteDiscount() {
      const qc = useQueryClient();
      return useMutation({
        mutationFn: deleteDiscount,
        onSuccess: () => {
          qc.invalidateQueries({ queryKey: ['deal-discounts'] });
          qc.invalidateQueries({ queryKey: ['deal-audit-log'] });
        },
      });
    }

              type="number"
              step="0.01"
              {...register('discountPercentage', { valueAsNumber: true })}
              disabled={pctDisabled}
              placeholder="Korting %"

    import { useState } from 'react';
    import { useDealDiscounts, useCreateDiscount, useUpdateDiscount, useDeleteDiscount } from '../hooks/useDealDiscounts';
    import { DiscountRow } from './DiscountRow';
    import type { DealOutcome } from '../types';
    import { useModules } from '@/features/admin/hooks/useModules'; // or wherever module list comes from

    interface DiscountEditorProps {
      dealOutcome: DealOutcome;
    }

    export function DiscountEditor({ dealOutcome }: DiscountEditorProps) {
      const { data: discounts = [] } = useDealDiscounts(dealOutcome.id);
      const { mutate: create } = useCreateDiscount();
      const { mutate: update } = useUpdateDiscount();
      const { mutate: del } = useDeleteDiscount();
      const [addingRow, setAddingRow] = useState(false);
      const [editingId, setEditingId] = useState<string | null>(null);
      const modules = useModules(); // canonical module list

      // Group by (module, provider) for quick lookup

      return (
        <section className="bg-white rounded-lg border border-neutral-200 p-6">
          <header className="flex items-center justify-between mb-4">

            {!addingRow && (
              <button
                type="button"
                onClick={() => setAddingRow(true)}
                className="h-11 px-4 text-cito-primary text-[14px] font-semibold border border-cito-primary rounded-lg"
              >
                + Korting toevoegen
              </button>
            )}
          </header>

            <div className="text-center py-8">
              <p className="text-[14px] text-neutral-700 mb-2 font-semibold">Nog geen kortingen vastgelegd</p>
              <p className="text-[14px] text-neutral-500">Voeg per module en aanbieder een korting toe (in % of € — niet beide tegelijk).</p>
            </div>

          {/* Existing discounts list */}
          <ul className="space-y-2">
            {discounts.map((d) => {
              const mod = modules.find((m) => m.id === d.moduleId);
              if (!mod) return null;
              if (editingId === d.id) {
                return (
                  <li key={d.id}>
                    <DiscountRow
                      module={mod}
                      provider={d.provider}
                      initial={{ discountPercentage: d.discountPercentage, discountAmount: d.discountAmount }}
                      onSave={(data) => {
                        update({ id: d.id, patch: { discountPercentage: data.discountPercentage, discountAmount: data.discountAmount } });
                        setEditingId(null);
                      }}
                      onCancel={() => setEditingId(null)}
                      onDelete={() => { del(d.id); setEditingId(null); }}
                    />
                  </li>
                );
              }
              return (
                <li key={d.id} className="flex items-center justify-between p-3 bg-neutral-50 rounded-lg min-h-[44px]">
                  <span className="text-[14px]">
                    <strong>{mod.label}</strong> · {d.provider.toUpperCase()}
                  </span>
                  <span className="text-[14px] tabular-nums">
                    {d.discountPercentage !== undefined && d.discountPercentage !== null
                      ? `${d.discountPercentage}%`
                      : `€ ${d.discountAmount?.toFixed(2) ?? '0,00'}`}
                  </span>
                  <div className="flex items-center gap-2">
                    <button type="button" onClick={() => setEditingId(d.id)} aria-label={`Korting bewerken voor ${mod.label} (${d.provider})`} className="w-11 h-11 flex items-center justify-center">
                      <svg className="w-4 h-4 text-cito-primary" viewBox="0 0 16 16"><path d="..." /></svg>
                      <span className="sr-only">Bewerken</span>
                    </button>
                    <button type="button" onClick={() => del(d.id)} aria-label={`Korting verwijderen voor ${mod.label} (${d.provider})`} className="w-11 h-11 flex items-center justify-center">
                      <svg className="w-4 h-4 text-red-600" viewBox="0 0 16 16"><path d="..." /></svg>
                      <span className="sr-only">Verwijderen</span>
                    </button>
                  </div>
                </li>
              );
            })}
          </ul>

          {/* New row */}
          {addingRow && (
            <NewDiscountRow
              dealOutcomeId={dealOutcome.id}
              existingKeys={new Set(byKey.keys())}
              onSave={(data) => { create({ dealOutcomeId: dealOutcome.id, ...data }); setAddingRow(false); }}
              onCancel={() => setAddingRow(false)}
              modules={modules}
            />
          )}
        </section>
      );
    }
