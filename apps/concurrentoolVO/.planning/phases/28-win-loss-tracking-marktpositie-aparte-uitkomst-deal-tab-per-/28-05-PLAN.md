    import { supabase } from '@/lib/supabase/client';
    import type { DealOutcome, ComparisonSnapshot } from '@/features/deal-outcomes/types';
    import { mapDealOutcomeRow, mapDealOutcomeInsert } from '@/db/types';

    /** Fetch the active (non-archived) deal_outcomes row for a school, if any. */
    export async function fetchActiveDealOutcome(schoolId: string): Promise<DealOutcome | null> {
      const { data, error } = await supabase
        .from('deal_outcomes')
        .select('*')
        .eq('school_id', schoolId)
        .neq('status', 'archived')
        .order('created_at', { ascending: false })
        .limit(1)
        .maybeSingle();
      if (error) throw error;
      return data ? mapDealOutcomeRow(data) : null;
    }

    /** Create a new deal outcome (typically status='open'/'won'/'lost').
     *  Writes audit-log entry 'outcome_created' on success. */
    export async function createDealOutcome(input: {
      schoolId: string;
      teamId: string;
      status: DealStatus;
      competitorProvider: 'dia' | 'jij' | 'overig';
      competitorName?: string;
      reason?: string;
      reasonCategory?: ReasonCategory;
      contactId?: string;
      comparisonSnapshot: ComparisonSnapshot;
      decidedAt?: string;
    }): Promise<DealOutcome> {
      // If status is won/lost, archive any current active record first
      if (input.status === 'won' || input.status === 'lost') {
        await archiveActiveDealOutcomes(input.schoolId);
      }
      const { data, error } = await supabase
        .from('deal_outcomes')
        .insert(mapDealOutcomeInsert(input as any))
        .select('*')
        .single();
      if (error) throw error;
      const outcome = mapDealOutcomeRow(data);
      // Audit-log (defensive — failure does not roll back primary write)
      try {
        await supabase.from('deal_audit_log').insert({
          deal_outcome_id: outcome.id,
          action: 'outcome_created',
          entity_type: 'deal_outcome',
          entity_id: outcome.id,
          before_value: null,
          after_value: { status: outcome.status, competitor_provider: outcome.competitorProvider },
        });
      } catch (auditErr) {
        console.warn('[deal-outcomes-operations] audit-log write failed', auditErr);
      }
      return outcome;
    }

    /** Update an existing deal outcome. Writes audit-log entry 'outcome_updated' or 'status_changed'. */
    export async function updateDealOutcome(id: string, patch: Partial<DealOutcome>): Promise<DealOutcome> {
      // Fetch before-value for audit
      const { data: before } = await supabase.from('deal_outcomes').select('*').eq('id', id).single();
      const updatePatch = mapDealOutcomeInsert(patch as any);
      const { data, error } = await supabase
        .from('deal_outcomes')
        .update(updatePatch)
        .eq('id', id)
        .select('*')
        .single();
      if (error) throw error;
      const after = mapDealOutcomeRow(data);
      try {
        const isStatusChange = before && before.status !== data.status;
        await supabase.from('deal_audit_log').insert({
          deal_outcome_id: id,
          action: isStatusChange ? 'status_changed' : 'outcome_updated',
          entity_type: 'deal_outcome',
          entity_id: id,
          before_value: before,
          after_value: data,
        });
      } catch (auditErr) {
        console.warn('[deal-outcomes-operations] audit-log write failed', auditErr);
      }
      return after;
    }

    /** Archive all currently-active deal_outcomes for a school (used before creating a new one). */
    async function archiveActiveDealOutcomes(schoolId: string): Promise<void> {
      const { error } = await supabase
        .from('deal_outcomes')
        .update({ status: 'archived' })
        .eq('school_id', schoolId)
        .in('status', ['open', 'in_negotiation']);
      if (error) throw error;
    }

    /** Explicitly archive a specific deal_outcome (e.g. "Heropen deal" flow per UI-SPEC). */
    export async function archiveDealOutcome(id: string): Promise<void> {
      await updateDealOutcome(id, { status: 'archived' });
    }

    import type { ComparisonSnapshot } from '@/features/deal-outcomes/types';
    import { useSchoolProfileStore } from '@/features/school-profile/store'; // verify exact import path
    import { calculateComparison, type ProviderKey } from '@/engine/price-comparison';
    import { MODULE_CATALOG } from '@/models/modules';

    /**
     * Phase 28 R1: build a frozen ComparisonSnapshot from current school state.
     * Reads store via getState() per CLAUDE.md gotcha (NOT via hooks — avoids stale closures).
     *
     * @param schoolId            ID of the school whose state we snapshot
     * @param competitorProvider  Pre-resolved competitor ('dia'|'jij'|'overig') from wizard/Uitkomst-tab.
     *                            If user has not chosen, caller invokes resolveCompetitorProvider()
     *                            below which defaults to the cheapest non-cito provider.
     * @returns ComparisonSnapshot matching comparisonSnapshotSchema (Plan 01)
     * @throws Error if school has no selected modules — caller must block dialog open.
     */
    export function computeComparisonSnapshot(
      schoolId: string,
      competitorProvider: 'dia' | 'jij' | 'overig',
    ): ComparisonSnapshot {
      // CLAUDE.md gotcha: getState pattern — NOT useSchoolProfileStore() hook
      const store = useSchoolProfileStore.getState();
      const profile = store.schools?.[schoolId] ?? store; // adjust to actual store shape
      const selectedModules = profile.selectedModules ?? [];
      const studentCounts = profile.studentCounts ?? {};
      const scenario = profile.scenario;

      if (selectedModules.length === 0) {
        throw new Error('Geen modules geselecteerd — vul eerst het schoolprofiel aan.');
      }

      // Engine doesn't know "overig" — map it to the cheapest non-cito provider for the
      // snapshot total. The literal "overig" stays in the JSONB for UI display fidelity.
      const result = calculateComparison(selectedModules, studentCounts, { scenarioType: scenario });
      const engineCompetitor: ProviderKey =
        competitorProvider === 'overig'
          ? cheapestNonCito(result)
          : competitorProvider;

      const citoTotal = result.totals.cito;
      const competitorTotal = result.totals[engineCompetitor];
      const difference = citoTotal - competitorTotal;

      const perModuleBreakdown = result.modules.map((m) => {
        const moduleDef = MODULE_CATALOG.find((md) => md.id === m.moduleId);
        return {
          moduleId: m.moduleId,
          moduleLabel: moduleDef?.name ?? m.moduleId,
          citoPricePerStudent: m.providers.cito?.pricePerStudent ?? 0,
          competitorPricePerStudent: m.providers[engineCompetitor]?.pricePerStudent ?? 0,
        };
      });

      const providersInScope: ProviderKey[] = (['cito', 'dia', 'jij', 'saqi'] as const).filter(
        (p) => result.modules.some((m) => m.providers[p] !== null),
      );

      return {
        citoTotal,
        competitorProvider, // keep the literal user choice (incl. 'overig')
        competitorTotal,
        difference,
        perModuleBreakdown,
        providersInScope,
        snapshotAt: new Date().toISOString(),
      };
    }

    /** Pick the cheapest non-cito provider in totals; default to 'dia' if no data. */
    function cheapestNonCito(result: ReturnType<typeof calculateComparison>): ProviderKey {
      const candidates: Array<[ProviderKey, number]> = [
        ['dia', result.totals.dia],
        ['jij', result.totals.jij],
        ['saqi', result.totals.saqi],
      ];
      const valid = candidates.filter(([, t]) => t > 0);
      if (valid.length === 0) return 'dia';
      return valid.reduce((min, cur) => (cur[1] < min[1] ? cur : min))[0];
    }

    /** Convenience for callers: resolve form value → 'dia'|'jij'|'overig' with fallback. */
    export function resolveCompetitorProvider(
      formValue: string | undefined,
    ): 'dia' | 'jij' | 'overig' {
      if (formValue === 'dia' || formValue === 'jij' || formValue === 'overig') return formValue;
      // Default: 'dia' (most common NL VO competitor per RESEARCH). Caller can override
      // by reading totals via cheapestNonCito() if a real ComparisonResult is available.
      return 'dia';
    }
