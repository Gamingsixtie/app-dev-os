    **`src/features/deal-outcomes/hooks/useCohortPrediction.ts`** — Matview lookup (per RESEARCH Code Example):
    ```typescript
    import { useQuery } from '@tanstack/react-query';
    import { supabase } from '@/lib/supabase/client';
    import { mapCohortStatsRow } from '@/db/types';
    import type { Onderwijsvisie } from '@/features/deal-outcomes/types';
    import type { SchoolLevel } from '@/models/school';

    interface CohortLookupInput {
      teamId?: string;
      onderwijsvisie?: Onderwijsvisie;
      primaryLevel?: SchoolLevel;
    }

    export function useCohortPrediction({ teamId, onderwijsvisie, primaryLevel }: CohortLookupInput) {
      return useQuery({
        queryKey: ['cohort-prediction', teamId, onderwijsvisie, primaryLevel],
        queryFn: async () => {
          if (!teamId || !onderwijsvisie || !primaryLevel) return null;
          const { data, error } = await supabase
            .from('deal_cohort_stats')
            .select('team_id, onderwijsvisie, primary_level, total_deals, won_deals, lost_deals, win_rate, top_lost_reason')
            .eq('team_id', teamId)
            .eq('onderwijsvisie', onderwijsvisie)
            .eq('primary_level', primaryLevel)
            .maybeSingle();
          if (error) throw error;
          return data ? mapCohortStatsRow(data) : null; // null = N=0 cohort
        },
        enabled: !!(teamId && onderwijsvisie && primaryLevel),
        staleTime: 5 * 60 * 1000, // 5min — matview refreshes on deal-mutation
      });
    }

    /** Helper to pick primary level for a school with potentially multiple levels.
     *  Per user-confirmed decision: use sub-type granularity, take levels[0] as primary. */
    export function resolvePrimaryLevel(levels: SchoolLevel[] | undefined): SchoolLevel | undefined {
      return levels && levels.length > 0 ? levels[0] : undefined;
    }
    ```

    **`src/features/deal-outcomes/components/CohortPredictionCard.tsx`** — 4-state render (per UI-SPEC §Copywriting Contract cohort-prediction templates + D-11):
    ```tsx
    import { Link } from '@tanstack/react-router';
    import { useCohortPrediction, resolvePrimaryLevel } from '../hooks/useCohortPrediction';
    import { ONDERWIJSVISIE_LABELS, REASON_CATEGORY_LABELS } from '../labels';
    import type { DealStatus, Onderwijsvisie } from '../types';
    import type { SchoolRecord } from '@/db/types';

    interface CohortPredictionCardProps {
      school: SchoolRecord;
      status: DealStatus;
    }

    export function CohortPredictionCard({ school, status }: CohortPredictionCardProps) {
      // Per D-11: hide for completed deals
      if (status === 'won' || status === 'lost' || status === 'archived') return null;

      const primaryLevel = resolvePrimaryLevel(school.levels);
      const onderwijsvisie = school.onderwijsvisie as Onderwijsvisie | undefined;

      // State 1: missing cohort features
      if (!onderwijsvisie || !primaryLevel) {
        return (
          <section className="bg-blue-50 border border-blue-200 rounded-lg p-6" role="status">
            <h2 className="text-[20px] font-semibold text-blue-800 mb-2">Onvoldoende schoolgegevens voor voorspelling</h2>
            <p className="text-[14px] text-blue-800 mb-4">
              Vul onderwijsvisie en schoolniveau aan op het schoolprofiel om een voorspelling te zien.
            </p>
            <Link
              to="/scholen/$slug"
              params={{ slug: school.slug }}
              className="inline-flex items-center h-11 px-4 text-blue-800 text-[14px] font-semibold border border-blue-300 rounded-lg"
            >
              Naar schoolprofiel
            </Link>
          </section>
        );
      }

      const { data: cohort, isLoading } = useCohortPrediction({
        teamId: school.teamId,
        onderwijsvisie,
        primaryLevel,
      });

      if (isLoading) {
        return <div className="h-32 bg-neutral-200 animate-pulse rounded-lg" />;
      }

      // State 2: N=0 cohort
      if (!cohort || cohort.totalDeals === 0) {
        return (
          <section className="bg-blue-50 border border-blue-200 rounded-lg p-6" role="status">
            <h2 className="text-[20px] font-semibold text-blue-800 mb-2">Eerste in zijn cohort</h2>
            <p className="text-[14px] text-blue-800">
              Er zijn nog geen vergelijkbare scholen met deze combinatie van onderwijsvisie en niveau.
              Zodra collega's vergelijkbare deals registreren, ziet u hier een voorspelling.
            </p>
          </section>
        );
      }

      // Cohort exists — compute display
      const onderwijsvisieLabel = ONDERWIJSVISIE_LABELS[onderwijsvisie];
      const winRatePct = cohort.winRate ?? 0;
      const lowConfidence = cohort.totalDeals < 5;
      const topReasonLabel = cohort.topLostReason
        ? REASON_CATEGORY_LABELS[cohort.topLostReason]
        : undefined;

      return (
        <details
          open
          className={`rounded-lg border p-6 ${lowConfidence ? 'bg-orange-50 border-orange-200' : 'bg-white border-neutral-200'}`}
        >
          <summary className="cursor-pointer text-[20px] font-semibold text-cito-primary mb-2 list-none flex items-center justify-between">
            <span>Voorspelling — cohort-gebaseerd</span>
            <span className="text-[12px] text-neutral-500">Vouw uit/in</span>
          </summary>
          <div className="mt-4">
            <p className="text-[16px] text-neutral-700 mb-2">
              Vergelijkbaar met <strong>{cohort.totalDeals} {onderwijsvisieLabel}-{primaryLevel}-scholen</strong>
              {' '}— win-kans <strong>{winRatePct.toFixed(0)}%</strong>.
            </p>

            {/* State 3: low confidence (1-4 deals) */}
            {lowConfidence && (
              <p role="status" className="text-[14px] text-orange-800 mb-2">
                ⚠ Lage betrouwbaarheid — slechts {cohort.totalDeals} vergelijkbare scholen.
              </p>
            )}

            {/* State 4: top lost reason (when cohort has lost deals) */}
            {topReasonLabel && cohort.lostDeals > 0 && (
              <p className="text-[14px] text-neutral-700">
                In dit cohort gaan deals het vaakst verloren op <strong>{topReasonLabel.toLowerCase()}</strong>.
              </p>
            )}

            {/* Matching features explainer */}
            <p className="text-[12px] text-neutral-500 mt-3">
              Gebaseerd op: onderwijsvisie ({onderwijsvisieLabel}) + niveau ({primaryLevel}).
            </p>
          </div>
        </details>
      );
    }
    ```
