    ```sql
    -- Phase 28 R4 / B6 fix: server-side aggregation for dashboard KPIs.
    -- Single round-trip; correct date-range + level filter semantics.
    CREATE OR REPLACE FUNCTION public.get_deal_stats(
      p_team_id    uuid,
      p_period_from timestamptz,
      p_period_to   timestamptz,
      p_levels      text[]      -- NULL or empty = all levels
    )
    RETURNS TABLE (
      total_deals    integer,
      won_deals      integer,
      lost_deals     integer,
      open_deals     integer,
      win_rate       numeric(5,4),       -- 0..1 or NULL when no decided deals
      avg_margin_eur numeric(12,2)        -- NULL when no margins available
    )
    LANGUAGE sql
    STABLE
    SECURITY INVOKER  -- uses caller's RLS context — defense in depth on team_id
    AS $$
      WITH filtered AS (
        SELECT
          d.id,
          d.status,
          d.comparison_snapshot,
          d.decided_at,
          d.created_at,
          s.levels AS school_levels
        FROM deal_outcomes d
        JOIN schools s ON s.id = d.school_id
        WHERE d.team_id = p_team_id
          AND d.status <> 'archived'
          AND (
            -- Decided in period
            (d.decided_at IS NOT NULL AND d.decided_at >= p_period_from AND d.decided_at <= p_period_to)
            OR
            -- Undecided but created in period
            (d.decided_at IS NULL AND d.created_at >= p_period_from AND d.created_at <= p_period_to)
          )
          AND (
            -- Level filter: NULL/empty = pass-through; otherwise require overlap with school's levels JSONB array
            p_levels IS NULL
            OR cardinality(p_levels) = 0
            OR EXISTS (
              SELECT 1
              FROM jsonb_array_elements_text(s.levels) AS lvl
              WHERE lvl = ANY (p_levels)
            )
          )
      ),
      counts AS (
        SELECT
          COUNT(*)::integer                                                AS total_deals,
          COUNT(*) FILTER (WHERE status = 'won')::integer                  AS won_deals,
          COUNT(*) FILTER (WHERE status = 'lost')::integer                 AS lost_deals,
          COUNT(*) FILTER (WHERE status IN ('open','in_negotiation'))::integer AS open_deals,
          NULLIF(
            COUNT(*) FILTER (WHERE status IN ('won','lost'))::numeric,
            0
          )                                                                AS decided_n,
          COUNT(*) FILTER (WHERE status = 'won')::numeric                  AS won_n,
          AVG((comparison_snapshot->>'difference')::numeric)
            FILTER (WHERE (comparison_snapshot->>'difference') IS NOT NULL) AS avg_margin
        FROM filtered
      )
      SELECT
        total_deals,
        won_deals,
        lost_deals,
        open_deals,
        CASE WHEN decided_n IS NOT NULL THEN ROUND(won_n / decided_n, 4) ELSE NULL END AS win_rate,
        ROUND(avg_margin, 2) AS avg_margin_eur
      FROM counts;
    $$;

    GRANT EXECUTE ON FUNCTION public.get_deal_stats(uuid, timestamptz, timestamptz, text[]) TO authenticated;
    ```

    **Note on schools.levels JSONB shape:** This RPC assumes `schools.levels` is a JSONB array
    of strings like `["vmbo-b","havo"]`. If the column is text[] or a different shape, adjust
    the EXISTS subquery accordingly (Plan 02 ships the canonical schema; cross-check before
    applying).

    **Trust boundary (per CONTEXT D-22):** `SECURITY INVOKER` keeps RLS in caller's context.
    The explicit `p_team_id` parameter is defense-in-depth — RLS on deal_outcomes already
    scopes by team_id, so the function cannot leak across teams even if a caller passed
    a forged p_team_id (RLS would filter the underlying SELECT to 0 rows).

    ```typescript
    import { z } from 'zod';

    export const dashboardSearchSchema = z.object({
      period: z.enum(['30d', '90d', '365d', 'custom']).default('90d'),
      level: z.enum(['alle', 'vmbo', 'vmbo-b', 'vmbo-k', 'vmbo-gt', 'havo', 'vwo']).default('alle'),
      trendMetric: z.enum(['count', 'winRate']).default('count'),
      trendGrouping: z.enum(['week', 'month', 'quarter']).default('month'),
      from: z.string().date().optional(),
      to: z.string().date().optional(),
    }).refine(
      (d) => d.period !== 'custom' || (d.from !== undefined && d.to !== undefined),
      { message: 'Bij custom-periode zijn from + to verplicht', path: ['from'] },
    ).refine(
      (d) => d.from === undefined || d.to === undefined || new Date(d.from) <= new Date(d.to),
      { message: 'from moet voor to liggen', path: ['from'] },
    );

    export type DashboardSearch = z.input<typeof dashboardSearchSchema>;
    export type DashboardSearchResolved = z.output<typeof dashboardSearchSchema>;

    /** Convert period enum to (from, to) date pair for SQL queries. */
    export function resolvePeriod(s: DashboardSearchResolved): { from: Date; to: Date } {
      const now = new Date();
      const days = s.period === '30d' ? 30 : s.period === '90d' ? 90 : 365;
      if (s.period === 'custom' && s.from && s.to) {
        return { from: new Date(s.from), to: new Date(s.to) };
      }
      return { from: new Date(now.getTime() - days * 86400000), to: now };
    }

    /** Resolve level filter to SQL-IN list (handles 'vmbo' rollup client-side). */
    export function resolveLevels(level: DashboardSearch['level']): string[] | null {
      if (level === 'alle') return null;
      if (level === 'vmbo') return ['vmbo-b', 'vmbo-k', 'vmbo-gt'];
      return [level];
    }
    ```

    **`src/features/dashboard/hooks/useDealStats.ts`** — KPI aggregation:
    ```typescript
    import { useQuery } from '@tanstack/react-query';
    import { supabase } from '@/lib/supabase/client';
    import { resolvePeriod, resolveLevels, type DashboardSearchResolved } from '../types';
    import { useAuth } from '@/features/auth/useAuth'; // verify path

    export interface DealStats {
      totalDeals: number;
      winRate: number | null; // 0-1 or null if no won/lost deals
      avgMarginEur: number | null;
      dealsThisPeriod: number;
    }

    export function useDealStats(filter: DashboardSearchResolved) {
      const { teamId } = useAuth(); // assumes auth-hook exposes teamId
      const { from, to } = resolvePeriod(filter);
      const levels = resolveLevels(filter.level);

      return useQuery({
        queryKey: ['deal-stats', teamId, filter.period, filter.level, filter.from, filter.to],
        queryFn: async (): Promise<DealStats> => {
          if (!teamId) return { totalDeals: 0, winRate: null, avgMarginEur: null, dealsThisPeriod: 0 };

          // B6 FIX: Call server-side RPC instead of buggy `.or()` + `.lte()` chain.
          // Migration 019 defines get_deal_stats(team_id, period_from, period_to, level_filter)
          // which returns { total_deals, won_deals, lost_deals, win_rate, avg_margin_eur }.
          // The RPC handles the decided_at-vs-created_at branching server-side AND filters
          // schools.levels via JSONB containment server-side (faster, correct).
          const { data, error } = await supabase.rpc('get_deal_stats', {
            p_team_id: teamId,
            p_period_from: from.toISOString(),
            p_period_to: to.toISOString(),
            p_levels: levels, // string[] | null (null = all levels)
          });
          if (error) throw error;

          // RPC returns a single-row result set (or empty if no rows)
          const row = Array.isArray(data) && data.length > 0 ? data[0] : null;
          if (!row) {
            return { totalDeals: 0, winRate: null, avgMarginEur: null, dealsThisPeriod: 0 };
          }
          return {
            totalDeals: row.total_deals ?? 0,
            winRate: row.win_rate, // null when no decided deals
            avgMarginEur: row.avg_margin_eur, // null when no margins
            dealsThisPeriod: row.total_deals ?? 0,
          };
        },
        enabled: !!teamId,
        staleTime: 60 * 1000,
      });
    }
    ```
