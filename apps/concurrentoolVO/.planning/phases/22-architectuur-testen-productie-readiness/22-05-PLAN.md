1. Install Playwright: `npm install -D @playwright/test`
2. Install Chromium browser: `npx playwright install chromium`

3. Create `playwright.config.ts`:
```typescript
import { defineConfig, devices } from '@playwright/test';

export default defineConfig({
  testDir: './e2e',
  fullyParallel: true,
  forbidOnly: !!process.env.CI,
  retries: process.env.CI ? 2 : 0,
  workers: process.env.CI ? 1 : undefined,
  reporter: process.env.CI ? 'github' : 'html',
  use: {
    baseURL: 'http://localhost:5173',
    trace: 'on-first-retry',
    screenshot: 'only-on-failure',
  },
  projects: [
    { name: 'chromium', use: { ...devices['Desktop Chrome'] } },
  ],
  webServer: {
    command: 'npm run dev',
    url: 'http://localhost:5173',
    reuseExistingServer: !process.env.CI,
    timeout: 120_000,
    env: {
      SKIP_AUTH: 'true',
    },
  },
});
```

4. Create `e2e/helpers/auth.ts`:
```typescript
// Auth is bypassed via SKIP_AUTH=true in webServer env
// This helper provides any auth-related test utilities if needed
export async function ensureLoggedIn(page: import('@playwright/test').Page) {
  // With SKIP_AUTH=true, no login needed
  // If auth wall appears anyway, handle gracefully
  const loginButton = page.getByRole('button', { name: /inloggen/i });
  if (await loginButton.isVisible({ timeout: 1000 }).catch(() => false)) {
    throw new Error('Auth wall visible despite SKIP_AUTH=true');
  }
}
```

5. Create `e2e/helpers/school.ts`:
```typescript
import { type Page } from '@playwright/test';

export async function createSchool(page: Page, name: string) {
  await page.getByRole('button', { name: /school toevoegen/i }).click();
  await page.getByLabel(/schoolnaam/i).fill(name);
  await page.getByRole('button', { name: /aanmaken/i }).click();
}
```

6. Add to `package.json` scripts:
```json
"test:e2e": "npx playwright test",
"test:e2e:ui": "npx playwright test --ui"
```

7. Add to `.gitignore`:
