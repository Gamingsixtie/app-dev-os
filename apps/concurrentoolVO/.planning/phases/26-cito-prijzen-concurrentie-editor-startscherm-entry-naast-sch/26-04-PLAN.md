<objective>
Lever een AI-gestuurde Excel-import voor prijs-updates op `/prijzen`. Gebruiker kiest provider, upload `.xlsx`, frontend parseert via `xlsx`, Vercel function `api/ai-price-import.ts` laat `claude-haiku-4-5` de rijen mappen naar het `PROVIDER_CONFIGS[provider]` shape via een Zod-schema. Resultaat is een **per-rij diff-view** (huidig → voorgesteld) met checkbox-acceptatie. Pas na expliciete bevestiging wordt geschreven naar Supabase via de bestaande `updatePricingConfig` pipeline.

Purpose: Excel is de huidige werkrealiteit voor prijslijst-updates van concurrenten en Cito-aanbiedingen. Handmatig overtypen in de prijs-editor is foutgevoelig en traag. AI doet de mapping; de mens bewaakt de kwaliteit via de diff-view.

Output:
- Nieuwe Vercel function `api/ai-price-import.ts` (mirror van `api/ai-intake.ts`)
- Zod schemas per provider (single source of truth voor server + client)
- Pure functions: `parseExcelToRows`, `computePriceDiff`, `applyAcceptedDiff`
- 3 React components: Dropzone, DiffView, Flow (orchestrator/modaal)
- Tests voor pure logica (parser, diff, schemas)
- Integratie in `PrijzenPage` als knop "Importeer prijzen uit Excel"

NOT integrated with `ops-competitor-intel` skill (D-07 — accept duplicate-code risk in ruil voor focused single-purpose flow).
NOT a streaming AI-call — single request/response with Zod-validated JSON output is enough voor batch-imports.
Single-provider per upload (D-11) — minder AI ambiguity, kleinere prompt.
</objective>

       ```ts
       import * as XLSX from 'xlsx';
       
       const MAX_FILE_BYTES = 5 * 1024 * 1024; // 5MB cap (D-defensive)
       const ACCEPTED_EXT = '.xlsx';
       
       export interface ExcelValidationResult {
         valid: boolean;
         error?: string;
       }
       
       export function validateExcelInput(file: File | null | undefined): ExcelValidationResult {
         if (!file) return { valid: false, error: 'Geen bestand geselecteerd' };
         if (!file.name.toLowerCase().endsWith(ACCEPTED_EXT)) {
           return { valid: false, error: `Alleen .xlsx-bestanden worden ondersteund (kreeg ${file.name})` };
         }
         if (file.size > MAX_FILE_BYTES) {
           return { valid: false, error: `Bestand is te groot (max 5MB)` };
         }
         return { valid: true };
       }
       
       /** Reads first sheet, returns matrix of stringified cells. First row = headers. Pure on input. */
       export function parseExcelToRows(buffer: ArrayBuffer): string[][] {
         let wb: XLSX.WorkBook;
         try {
           wb = XLSX.read(buffer, { type: 'array' });
         } catch {
           throw new Error('Excel-bestand kon niet worden gelezen — is het wel een geldig .xlsx-bestand?');
         }
         const firstSheetName = wb.SheetNames[0];
         if (!firstSheetName) throw new Error('Excel-bestand bevat geen tabbladen');
         const sheet = wb.Sheets[firstSheetName];
         const rows = XLSX.utils.sheet_to_json<unknown[]>(sheet, { header: 1, defval: '' }) as unknown[][];
         if (rows.length === 0) throw new Error('Tabblad is leeg');
         return rows.map((row) => row.map((cell) => (cell == null ? '' : String(cell))));
       }
       ```

       ```ts
       export interface DiffRow {
         path: string;
         currentValue: unknown;
         proposedValue: unknown;
         changed: boolean;
       }
       
       /** Deep-walk both objects in parallel, emit one DiffRow per primitive leaf. */
       export function computePriceDiff(
         current: unknown,
         proposed: unknown,
         pathPrefix = '',
       ): DiffRow[] {
         const rows: DiffRow[] = [];
         if (typeof current !== 'object' || current === null || typeof proposed !== 'object' || proposed === null) {
           rows.push({
             path: pathPrefix || '$',
             currentValue: current,
             proposedValue: proposed,
             changed: !deepEqual(current, proposed),
           });
           return rows;
         }
         const cur = current as Record<string, unknown>;
         const prop = proposed as Record<string, unknown>;
         const keys = new Set([...Object.keys(cur), ...Object.keys(prop)]);
         for (const k of keys) {
           const subPath = pathPrefix ? `${pathPrefix}.${k}` : k;
           const cVal = cur[k];
           const pVal = prop[k];
           if (isPrimitive(cVal) && isPrimitive(pVal)) {
             rows.push({
               path: subPath,
               currentValue: cVal,
               proposedValue: pVal,
               changed: cVal !== pVal,
             });
           } else if (Array.isArray(cVal) && Array.isArray(pVal)) {
             // Treat arrays as opaque: one DiffRow representing the whole array if different.
             rows.push({
               path: subPath,
               currentValue: cVal,
               proposedValue: pVal,
               changed: !deepEqual(cVal, pVal),
             });
           } else {
             rows.push(...computePriceDiff(cVal, pVal, subPath));
           }
         }
         return rows;
       }
       
       function isPrimitive(v: unknown): v is string | number | boolean | null | undefined {
         return v === null || ['string', 'number', 'boolean', 'undefined'].includes(typeof v);
       }
       
       function deepEqual(a: unknown, b: unknown): boolean {
         return JSON.stringify(a) === JSON.stringify(b);
       }
       
       /** Build new object: take `current` and overlay `proposed` ONLY at the accepted paths. Pure (returns new object). */
       export function applyAcceptedDiff(
         current: unknown,
         diff: DiffRow[],
         acceptedPaths: ReadonlySet<string>,
       ): unknown {
         // Start with a deep clone of current
         const result = JSON.parse(JSON.stringify(current)) as unknown;
         for (const row of diff) {
           if (!acceptedPaths.has(row.path)) continue;
           setByPath(result, row.path, row.proposedValue);
         }
         return result;
       }
       
       function setByPath(obj: unknown, path: string, value: unknown): void {
         const parts = path.split('.');
         let cursor = obj as Record<string, unknown>;
         for (let i = 0; i < parts.length - 1; i++) {
           const k = parts[i];
           if (typeof cursor[k] !== 'object' || cursor[k] === null) cursor[k] = {};
           cursor = cursor[k] as Record<string, unknown>;
         }
         cursor[parts[parts.length - 1]] = value;
       }
       ```

    ```ts
        if (!parsed.success) {
          return jsonError(400, `Ongeldig verzoek: ${parsed.error.issues[0]?.message ?? 'onbekend'}`);
        }
        const { provider, excelRows, currentConfig } = parsed.data;
    
        const userMessage = `Aanbieder: ${provider}
    Huidige config:
    \`\`\`json
    ${JSON.stringify(currentConfig, null, 2)}
    \`\`\`
    
    Excel-rijen (eerste rij = headers):
    \`\`\`json
    ${JSON.stringify(excelRows.slice(0, 200), null, 2)}
    \`\`\`
    
    Produceer { "proposed": <nieuwe config>, "notes": "<optioneel>" }.`;
    
        const message = await anthropic.messages.create({
          model: 'claude-haiku-4-5',
          max_tokens: 4096,
          system: SYSTEM_PROMPT,
          messages: [{ role: 'user', content: userMessage }],
        });
    
        // Extract text content (Anthropic SDK returns content blocks)
        const text = message.content
          .filter((c): c is { type: 'text'; text: string } => c.type === 'text')
          .map((c) => c.text)
          .join('');
    
        // Try direct JSON.parse, then strip fences, then first-brace fallback
        let aiResult: { proposed?: unknown; notes?: string };
        try {
          aiResult = JSON.parse(text);
        } catch {
          const fence = text.match(/```(?:json)?\\s*\\n?([\\s\\S]*?)\\n?\\s*```/);
          const candidate = fence ? fence[1] : extractFirstJsonObject(text);
          if (!candidate) {
            return jsonError(502, 'AI-antwoord kon niet worden verwerkt (geen geldig JSON)');
          }
          try {
            aiResult = JSON.parse(candidate);
          } catch {
            return jsonError(502, 'AI-antwoord kon niet worden verwerkt (JSON-parsefout)');
          }
        }
    
        if (!aiResult || typeof aiResult.proposed !== 'object') {
          return jsonError(502, 'AI-antwoord mist het verwachte "proposed" object');
        }
    
        return new Response(
          JSON.stringify({ proposed: aiResult.proposed, notes: aiResult.notes }),
          { status: 200, headers: { 'Content-Type': 'application/json' } },
        );
      } catch (err) {
        const msg = err instanceof Error ? err.message : 'Onbekende fout';
        return jsonError(500, `Er is een fout opgetreden bij de AI-import: ${msg}`);
      }
    }
    
    function jsonError(status: number, message: string): Response {
      return new Response(JSON.stringify({ error: message }), {
        status,
        headers: { 'Content-Type': 'application/json' },
      });
    }
    
    function extractFirstJsonObject(text: string): string | null {
      const first = text.indexOf('{');
      const last = text.lastIndexOf('}');
      if (first === -1 || last <= first) return null;
      return text.slice(first, last + 1);
    }
    
    export default POST;
    ```

    ```ts
    async function getAuthHeaders(): Promise<Record<string, string>> {
      const headers: Record<string, string> = { 'Content-Type': 'application/json' };
      if (import.meta.env.VITE_SKIP_AUTH === 'true') return headers;
      const { data: { session } } = await supabase.auth.getSession();
      if (!session) throw new Error('Niet ingelogd. Log opnieuw in om AI-functies te gebruiken.');
      headers['Authorization'] = `Bearer ${session.access_token}`;
      return headers;
    }
    
    export interface AiPriceImportResult {
      proposed: unknown;  // shape validated against PROVIDER_IMPORT_SCHEMAS[provider]
      notes?: string;
    }
    
    export async function importPricesFromExcel(
      provider: ProviderImportKey,
      excelRows: string[][],
      currentConfig: Record<string, unknown>,
    ): Promise<AiPriceImportResult> {
      const headers = await getAuthHeaders();
      const response = await fetch('/api/ai-price-import', {
        method: 'POST',
        headers,
        body: JSON.stringify({ provider, excelRows, currentConfig }),
      });
      if (!response.ok) {
        let msg = 'AI-import mislukt.';
        try {
          const body = (await response.json()) as { error?: string };
          if (body.error) msg = body.error;
        } catch { /* fall through */ }
        throw new Error(msg);
      }
      const body = (await response.json()) as { proposed?: unknown; notes?: string };
      if (!body.proposed) throw new Error('AI gaf geen voorstel terug.');
      // Validate AI output against canonical schema CLIENT-SIDE (server only does request envelope check)
      const schema = PROVIDER_IMPORT_SCHEMAS[provider];
      const parsed = schema.safeParse(body.proposed);
      if (!parsed.success) {
        const first = parsed.error.issues[0];
        throw new Error(`AI-voorstel voldoet niet aan het verwachte prijsmodel: ${first?.message ?? 'validatiefout'}`);
      }
      return { proposed: parsed.data, notes: body.notes };
    }
    ```

    ```tsx
    import { useQueryClient } from '@tanstack/react-query';
    import { PROVIDER_CONFIGS } from '@/data/providers/index';
    import { usePricingConfigs } from '@/hooks/usePricingConfigs';
    import { updatePricingConfig } from '@/db/pricing-operations';

    import { PriceImportDropzone } from './PriceImportDropzone';
    import { PriceImportDiffView } from './PriceImportDiffView';
    import type { ProviderImportKey } from '../import/price-import-schemas';
    
    interface PriceImportFlowProps {
      open: boolean;
      onClose: () => void;
    }
    
    type Stage = 'upload' | 'aiCall' | 'diff' | 'saving';
    
    export function PriceImportFlow({ open, onClose }: PriceImportFlowProps) {
      const queryClient = useQueryClient();
      const { data: dbConfigs } = usePricingConfigs();
      const [stage, setStage] = useState<Stage>('upload');
      const [error, setError] = useState<string | null>(null);
      const [provider, setProvider] = useState<ProviderImportKey | null>(null);
      const [diff, setDiff] = useState<DiffRow[]>([]);
      const [proposed, setProposed] = useState<unknown>(null);
      const [aiNotes, setAiNotes] = useState<string | undefined>();
    
      if (!open) return null;
    
      const reset = () => {
        setStage('upload'); setError(null); setProvider(null); setDiff([]); setProposed(null); setAiNotes(undefined);
      };
    
      const handleSubmit = async ({ provider: p, file }: { provider: ProviderImportKey; file: File }) => {
        setStage('aiCall');
        setError(null);
        setProvider(p);
        try {
          const buffer = await file.arrayBuffer();
          const rows = parseExcelToRows(buffer);
          const currentStrategy = PROVIDER_CONFIGS[p].pricingStrategy as unknown as Record<string, unknown>;
          const result = await importPricesFromExcel(p, rows, currentStrategy);
          const diffRows = computePriceDiff(currentStrategy, result.proposed);
          setDiff(diffRows);
          setProposed(result.proposed);
          setAiNotes(result.notes);
          setStage('diff');
        } catch (e) {
          setError(e instanceof Error ? e.message : 'Onverwachte fout');
          setStage('upload');
        }
      };
    
      const handleConfirm = async (acceptedPaths: ReadonlySet<string>) => {
        if (!provider) return;
        setStage('saving');
        setError(null);
        try {
          const currentStrategy = PROVIDER_CONFIGS[provider].pricingStrategy as unknown as Record<string, unknown>;
          const merged = applyAcceptedDiff(currentStrategy, diff, acceptedPaths) as Record<string, unknown>;
          const dbConfig = dbConfigs?.find((c) => c.provider === provider && c.is_active);
          if (!dbConfig) throw new Error(`Geen actieve DB-config voor ${provider} gevonden`);
          await updatePricingConfig(dbConfig.id, merged);
          await usePricingDataStore.getState().loadFromSupabase();
          queryClient.invalidateQueries({ queryKey: ['pricing-configs'] });
          reset();
          onClose();
        } catch (e) {
          setError(e instanceof Error ? e.message : 'Opslaan mislukt');
          setStage('diff');
        }
      };
    
      return (
        <div className="fixed inset-0 z-50 bg-black/50 flex items-center justify-center p-4" role="dialog" aria-modal="true">
          <div className="bg-white rounded-lg shadow-xl max-w-3xl w-full max-h-[90vh] overflow-y-auto p-6">

              <h2 className="text-lg font-semibold text-cito-primary">Importeer prijzen uit Excel</h2>

            )}
            {/* `proposed` kept in state for potential debug-display in future */}
            {proposed === null && null}
          </div>
        </div>
      );
    }
    ```
