       import type { PROVIDER_CONFIGS as ConfigsType } from '@/data/providers/index';
       import type { ProviderKey } from '@/engine/price-comparison';
       
       export const PRICE_LIST_DISCLAIMER =
         'Prijzen zijn indicatief en kunnen aangepast worden. Voor actuele bevestiging: contact Cito.';
       
       export interface PriceListSnapshotRow {
         provider: ProviderKey;
         providerLabel: string;
         pricingType: string;
         description: string;
       }
       
       export interface PriceListSnapshot {
         title: string;
         dateLabel: string;
         rows: PriceListSnapshotRow[];
         disclaimer: string;
       }
       
       const NL_MONTHS = [
         'januari','februari','maart','april','mei','juni',
         'juli','augustus','september','oktober','november','december',
       ];
       
       /** Format a Date as NL long format (e.g. "14 mei 2026"). Pure. */
       export function formatDateNL(d: Date): string {
         return `${d.getDate()} ${NL_MONTHS[d.getMonth()]} ${d.getFullYear()}`;
       }
       
       /** Build a flat view-model of the current PROVIDER_CONFIGS for export. */
       export function buildPriceListSnapshot(
         configs: typeof ConfigsType,
         now: Date = new Date(),
       ): PriceListSnapshot {
         const rows: PriceListSnapshotRow[] = [];
         for (const key of ['cito', 'dia', 'jij', 'saqi'] as ProviderKey[]) {
           const cfg = configs[key];
           if (!cfg) continue;
           const strat = cfg.pricingStrategy;
           let description = '';
           switch (strat.type) {
             case 'platform+module': {
               const bundles = strat.bundles
                 .filter((b) => b.pricePerStudent !== null)
                 .map((b) => `${b.name}: €${b.pricePerStudent?.toFixed(2)}/lln`)
                 .join(' • ');
               const individual = Object.entries(strat.individualPrices)
                 .map(([mod, price]) => `${mod} €${price.toFixed(2)}`)
                 .join(', ');
               description = `${bundles}. Individueel: ${individual}.`;
               break;
             }
             case 'package-bundle': {
               description = strat.packages
                 .map((p) => `${p.name}: €${p.pricePerStudent.toFixed(2)}/lln (modules: ${p.includedModuleIds.join(', ')})`)
                 .join(' • ');
               break;
             }
             case 'tiered-license': {
               description = strat.tiers
                 .map((t) => `${t.label}: €${t.annualFee.toFixed(2)}/jr + €${t.pricePerTest.toFixed(2)}/toets`)
                 .join(' • ');
               break;
             }
             case 'flat': {
               description = `€${strat.pricePerStudent.toFixed(2)}/lln`;
               break;
             }
             default:
               description = '(onbekend prijsmodel)';
           }
           rows.push({
             provider: key,
             providerLabel: cfg.label,
             pricingType: strat.type,
             description,
           });
         }
         return {
           title: 'Cito Rekentool — Prijslijst',
           dateLabel: formatDateNL(now),
           rows,
           disclaimer: PRICE_LIST_DISCLAIMER,
         };
       }
       ```

       import { renderPriceListHtml } from './price-list-html';
       import type { PriceListSnapshot } from './price-list-snapshot';
       
       /**
        * Renders a Word (.docx) Blob from the snapshot.
        * Uses html-docx-js (~50KB) to convert our HTML rendering to OOXML.
        * Library has no types; cast to a typed wrapper to keep the rest of the code clean.
        */
       export async function renderPriceListWordBlob(snapshot: PriceListSnapshot): Promise<Blob> {
         const html = renderPriceListHtml(snapshot);
         // Lazy import keeps the library out of the initial bundle.
         // Path is `html-docx-js/dist/html-docx` per the library's package layout.
         const mod = (await import('html-docx-js/dist/html-docx')) as unknown as {
           default?: { asBlob: (html: string) => Blob };
           asBlob?: (html: string) => Blob;
         };
         const lib = mod.default ?? mod;
         if (!lib.asBlob) {
           throw new Error('html-docx-js: asBlob niet beschikbaar — kan Word-bestand niet genereren.');
         }
         return lib.asBlob(html);
       }
       ```

       function downloadBlob(blob: Blob, filename: string) {
         const url = URL.createObjectURL(blob);
         const a = document.createElement('a');
         a.href = url;
         a.download = filename;
         a.click();
         URL.revokeObjectURL(url);
       }
       
       export function PriceListExportButton() {
         const [open, setOpen] = useState(false);
         const [busy, setBusy] = useState<Format | null>(null);
         const [error, setError] = useState<string | null>(null);
         const wrapRef = useRef<HTMLDivElement>(null);
       
         // close dropdown on outside click
         useEffect(() => {
           const handler = (e: MouseEvent) => {
             if (wrapRef.current && !wrapRef.current.contains(e.target as Node)) setOpen(false);

           document.addEventListener('mousedown', handler);
           return () => document.removeEventListener('mousedown', handler);

           setBusy(format);
           setError(null);

             const snapshot = buildPriceListSnapshot(PROVIDER_CONFIGS, new Date());

             if (format === 'txt') {

               const txt = renderPriceListTxt(snapshot);
               downloadBlob(new Blob([txt], { type: 'text/plain;charset=utf-8' }), `${baseName}.txt`);
             } else if (format === 'html') {

               downloadBlob(blob, `${baseName}.pdf`);

             setBusy(null);
           }

               type="button"
               onClick={() => setOpen((v) => !v)}
               disabled={busy !== null}
