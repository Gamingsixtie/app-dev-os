    ```ts
    import { useNavigate, useSearch } from '@tanstack/react-router';
    import type { PrijzenSearchParams, PrijzenTab, ConcurrentieSubTab } from '../types';
    
    /**
     * Reads + writes /prijzen URL search params (tab, provider).
     * Tab state is URL-backed for deeplinks + refresh-safety (D-02).
     */
    export function usePrijzenSearch() {
      // Type assertion: the prijzenRoute validateSearch guarantees this shape.
      const search = useSearch({ strict: false }) as PrijzenSearchParams;
      const navigate = useNavigate();
    
      const tab: PrijzenTab = search.tab ?? 'basis';
      const provider: ConcurrentieSubTab | undefined = search.provider;
    
      const setTab = (next: PrijzenTab) => {
        navigate({
          to: '/prijzen',
          search: (prev: Record<string, unknown>) => ({
            ...prev,
            tab: next,
            // When switching AWAY from 'concurrentie', clear the provider param so the URL stays clean.
            provider: next === 'concurrentie' ? (prev.provider as ConcurrentieSubTab | undefined) ?? 'dia' : undefined,
          }),
        });
      };
    
      const setProvider = (next: ConcurrentieSubTab) => {
        navigate({
          to: '/prijzen',
          search: (prev: Record<string, unknown>) => ({
            ...prev,
            tab: 'concurrentie',
            provider: next,
          }),
        });
      };
    
      return { tab, provider, setTab, setProvider };
    }
    ```

    ```tsx
    import { useState } from 'react';
    import { useQueryClient } from '@tanstack/react-query';
    import { PROVIDER_CONFIGS } from '@/data/providers/index';
    import { usePricingConfigs } from '@/hooks/usePricingConfigs';
    import { updatePricingConfig } from '@/db/pricing-operations';
    import { usePricingDataStore } from '@/stores/pricing-data-store';
    import { BASIS_MODULE_IDS } from '../constants/cito-module-grouping';
    import type { PlatformModulePricing } from '@/models/pricing';
    
    export function CitoBasisvaardigheidenTab() {
      const { data: dbConfigs } = usePricingConfigs();
      const queryClient = useQueryClient();
      const citoDbConfig = dbConfigs?.find((c) => c.provider === 'cito' && c.is_active);
      const citoStrategy = PROVIDER_CONFIGS.cito.pricingStrategy as PlatformModulePricing;
      const [localPrices, setLocalPrices] = useState<Record<string, number>>(() =>
        Object.fromEntries(
          Object.entries(citoStrategy.individualPrices).filter(([k]) => BASIS_MODULE_IDS.has(k))
        )
      );
      const [saving, setSaving] = useState(false);
      const [error, setError] = useState<string | null>(null);
      const [success, setSuccess] = useState(false);
    
      const handleSave = async () => {
        if (!citoDbConfig) {
          setError('Geen Cito-config gevonden in de database — kan niet opslaan.');
          return;
        }
        setSaving(true);
        setError(null);
        try {
          // Merge our edited basis-prices with the existing non-basis prices
          const mergedIndividualPrices = {
            ...citoStrategy.individualPrices,
            ...localPrices,
          };
          const newStrategy: PlatformModulePricing = {
            ...citoStrategy,
            individualPrices: mergedIndividualPrices,
          };
          await updatePricingConfig(citoDbConfig.id, newStrategy as unknown as Record<string, unknown>);
          await usePricingDataStore.getState().loadFromSupabase();
          queryClient.invalidateQueries({ queryKey: ['pricing-configs'] });
          setSuccess(true);
          setTimeout(() => setSuccess(false), 3000);
        } catch (e) {
          setError(e instanceof Error ? e.message : 'Opslaan mislukt');
        } finally {
          setSaving(false);
        }
      };
    
      return (
        <div className="space-y-4">
          <h3 className="text-lg font-semibold text-cito-primary">Cito Basisvaardigheden</h3>
          <p className="text-sm text-neutral-600">Prijs per leerling per jaar voor de basis-modules (rekenwiskunde, Nederlands, Engels).</p>
          <div className="space-y-2">
            {Array.from(BASIS_MODULE_IDS).map((moduleId) => (
              <div key={moduleId} className="flex items-center gap-3">
                <label className="text-sm text-gray-700 w-48 capitalize">{moduleId}</label>
                <input
                  type="number"
                  step="0.01"
                  value={localPrices[moduleId] ?? 0}
                  onChange={(e) =>
                    setLocalPrices({ ...localPrices, [moduleId]: parseFloat(e.target.value) || 0 })
                  }
                  className="w-32 rounded border border-gray-300 px-2 py-1 text-sm"
                />
                <span className="text-xs text-gray-500">€/lln</span>
              </div>
            ))}
          </div>
          {error && (
            <div className="bg-red-50 border border-red-200 rounded-md p-3 text-sm text-red-700">{error}</div>
          )}
          {success && (
            <div className="bg-green-50 border border-green-200 rounded-md p-3 text-sm text-green-700">Basisvaardigheid-prijzen opgeslagen</div>
          )}
          <button
            type="button"
            onClick={handleSave}
            disabled={saving}
            className="px-4 py-2 bg-cito-primary text-white rounded-md text-sm font-medium hover:bg-cito-primary/90 disabled:opacity-50"
          >
            {saving ? 'Opslaan...' : 'Basisvaardigheden opslaan'}
          </button>
        </div>
      );
    }
    ```
