```ts
// src/features/stichtingen/export/csvExport.ts
import * as XLSX from 'xlsx';

export interface StichtingCsvRow {
  schoolName: string;
  region: string;
  levels: string;          // comma-joined
  customerType: string;
  schoolType: string;
  growthTrajectory: string;
  pipelineStatus: string;
  totalStudents: number;
  modulesSelected: string; // comma-joined module-IDs
  currentToolMix: string;  // cito-only / concurrent-only / mixed
}

export function buildStichtingCsv(
  stichting: StichtingRecord,
  schools: SchoolRecord[],
): string {
  const rows: StichtingCsvRow[] = schools.map(toRow);
  const sheet = XLSX.utils.json_to_sheet(rows);
  return XLSX.utils.sheet_to_csv(sheet);
}
```
