         ```
         export function calculateBasisPlusUpsell(
           school: SchoolRecord,
           _options?: { bundle?: 'basis' | 'plus' },
         ): UpsellResult {
           // Pure function. Read CITO_BUNDLES.
           const basis = CITO_BUNDLES.find(b => b.id === 'basis');
           const plus = CITO_BUNDLES.find(b => b.id === 'plus');
           if (!basis || !plus) throw new Error('CITO_BUNDLES missing basis or plus');
           const deltaPerMonth = plus.pricePerStudent - basis.pricePerStudent;
           const totalStudents = Object.values(school.studentCounts ?? {}).flatMap(v => Object.values(v)).reduce((sum, n) => sum + (n ?? 0), 0);
           const deltaPerYear = deltaPerMonth * 12 * totalStudents;
           const additionalFeatures = plus.features.filter(f => !basis.features.includes(f));
           return { fromBundle: 'basis', toBundle: 'plus', deltaPerMonth, deltaPerYear, additionalFeatures };
         }
         ```
