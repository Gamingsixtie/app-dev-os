    ```typescript
    import { create } from 'zustand';
    import { calculateComparison } from '../../engine/price-comparison';
    import type { ComparisonResult } from '../../engine/price-comparison';
    import type { PriceRecord } from '../../models/pricing';
    import { DEFAULT_PRICES } from '../../data/default-prices';
    import { useSchoolProfileStore } from '../school-profile/store';

    export interface PriceOverride {
      moduleId: string;
      provider: 'cito' | 'dia' | 'jij';
      amount: number;
    }

    interface PriceComparisonState {
      result: ComparisonResult | null;
      draftOverrides: PriceOverride[];
      appliedOverrides: PriceOverride[];
      hasPendingChanges: boolean;

      initialize: () => void;
      setDraftOverride: (override: PriceOverride) => void;
      resetOverride: (moduleId: string, provider: string) => void;
      resetAllOverrides: () => void;
      recalculate: () => void;
    }
    ```

    Key implementation details:
    - `initialize()`: reads `useSchoolProfileStore.getState()` to get `selectedModules` and `studentCounts`. Calls `calculateComparison(selectedModules, studentCounts, DEFAULT_PRICES)`. Sets `result`.
    - `setDraftOverride(override)`: upserts into `draftOverrides` (replace if same moduleId+provider exists). Sets `hasPendingChanges = true`.
    - `resetOverride(moduleId, provider)`: removes from `draftOverrides`. Sets `hasPendingChanges = true`.
    - `resetAllOverrides()`: clears `draftOverrides` array. Sets `hasPendingChanges = true`.
    - `recalculate()`: builds merged price array: start with DEFAULT_PRICES, for each `appliedOverride` from `[...this.appliedOverrides, ...this.draftOverrides]`, create a new PriceRecord with overridden amountPerStudent, source='manual', sourceLabel='Handmatig ingevoerd'. Calls `calculateComparison` with merged prices. Moves `draftOverrides` to `appliedOverrides`. Sets `hasPendingChanges = false`.

    Export store as `usePriceComparisonStore`.
