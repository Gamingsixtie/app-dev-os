         }

           result: ComparisonResult,

           selectedModules: string[],
           moduleSetups: ModuleCurrentSetup[],
           schoolplanOpportunities?: Array<{ moduleId: string; kans: string }>,

             ...base,
             scenarioType: 'C' as const,

             },
           };
         }
